Pattern — public query with unstable_cache:

```typescript
import { unstable_cache } from "next/cache";
import { createClient } from "@/lib/db/supabase-server";

export const getPublications = unstable_cache(
  async (): Promise<Publication[]> => {
    const supabase = await createClient();
    const { data, error } = await supabase
      .from("publications")
      .select(PUB_SELECT)
      .eq("is_public", true) // VIS-01
      .neq("type", "patent")
      .order("year", { ascending: false });
    if (error) throw error;
    return (data ?? []).map(toPublication);
  },
  ["publications-public"], // cache key
  { tags: ["publications"] }, // invalidation tag
);
```

Pattern — internal query (no filter, no cache):

```typescript
export async function getAllPublications(): Promise<Publication[]> {
  const supabase = await createClient();
  const { data, error } = await supabase
    .from("publications")
    .select(PUB_SELECT)
    .order("year", { ascending: false });
  if (error) throw error;
  return (data ?? []).map(toPublication);
}
```
